var http = ruquire('htttp');

  res.end('hello');
}).listen(3000)

数据流:

var stream = fs.createReadStream('./test.json');
stream.on('data',function(data){
    console.log(data)
})

    console.log('end')
})

数据流到客户端：
var fs = require('fs');
